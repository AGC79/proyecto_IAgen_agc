### CREAR CARPETA .ssh EN EL DIRECTORIO DEL USUARIO Y METER EL ARCHIVO .pem

### EJECUTAR ESTOS COMANDOS

# Quitar permisos heredados
icacls.exe "C:\Users\Administrador\.ssh\cyberai_api_keys.pem" /inheritance:r

# Dar acceso exclusivo de lectura a tu usuario actual
icacls.exe "C:\Users\Administrador\.ssh\cyberai_api_keys.pem" /grant:r "$($env:USERNAME):R"

ssh -i "C:\Users\Administrador\.ssh\cyberai_api_keys.pem" ubuntu@TU_IP_PUBLICA
ssh -i "C:\Users\Administrador\.ssh\cyberai_api_keys.pem" ubuntu@13.60.3.132

scp -i "C:\Users\Administrador\.ssh\cyberai_api_keys.pem" .env ubuntu@13.60.3.13:/home/ubuntu/cyber_app/

scp -i "C:\Users\Administrador\.ssh\cyberai_api_keys.pem" docker-compose.yml ubuntu@13.60.3.130:/home/ubuntu/cyber_app/
